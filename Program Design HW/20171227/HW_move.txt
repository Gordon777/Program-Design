        {

            {
                if (this.Left >= pictureBox1.Left)
                {
                    a = 1;
                }
                else if (this.Right <= pictureBox1.Left + pictureBox1.Width)
                {
                    a = 0;
                }

                else if (a == 1)
                    pictureBox1.Left += 25;
